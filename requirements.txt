echo "📦 Распаковка..."
unzip -q "$ZIP_FILE" -d "$TMP_DIR" || { echo "❌ Ошибка распаковки. Проверьте путь к ZIP."; exit 1; }

echo "# 🧠 Контекст проекта" > "$OUT_FILE"
echo "" >> "$OUT_FILE"

echo "🔍 Фильтрация и сборка кода..."
find "$TMP_DIR" -type f \
  -not -path '*/node_modules/*' \
  -not -path '*/.git/*' \
  -not -path '*/venv/*' -not -path '*/.venv/*' \
  -not -path '*/__pycache__/*' -not -path '*/.mypy_cache/*' \
  -not -path '*/dist/*' -not -path '*/build/*' -not -path '*/.next/*' -not -path '*/out/*' \
  -not -path '*/.env*' -not -path '*/.env.*' \
  -not -path '*.pyc' -not -path '*.so' -not -path '*.lock' \
  \( -iname "*.py" -o -iname "*.js" -o -iname "*.jsx" -o -iname "*.ts" -o -iname "*.tsx" \
     -o -iname "*.vue" -o -iname "*.svelte" -o -iname "*.html" -o -iname "*.css" -o -iname "*.scss" \
     -o -iname "*.sql" -o -iname "*.yaml" -o -iname "*.yml" -o -iname "*.toml" -o -iname "*.json" \
     -o -iname "*.md" -o -iname "*.cfg" -o -iname "*.ini" -o -iname "Dockerfile" \
     -o -iname "docker-compose.yml" -o -iname "Makefile" \) \
  -print0 | while IFS= read -r -d '' file; do
    # Пропускаем бинарные файлы
    if file -b --mime-type "$file" | grep -q "^text/"; then
      rel="${file#$TMP_DIR/}"
      echo "### 📄 \`$rel\`" >> "$OUT_FILE"
      echo '
```' >> "<LaTex>id_2</LaTex>file" >> "<LaTex>id_1</LaTex>OUT_FILE"
    fi
  done

# Очистка временной папки
rm -rf "$TMP_DIR"

# Статистика
BYTES=$(wc -c < "$OUT_FILE")
WORDS=$(wc -w < "$OUT_FILE")
echo "✅ Готово: $OUT_FILE"
echo "📏 Размер: $(( BYTES / 1024 )) КБ"
echo "🧮 ~$(( BYTES / 4 )) токенов (приблизительно для кода)"
echo "💡 Если > 150К токенов — лучше разбить на backend / frontend отдельно"
EOF

chmod +x prepare_project.sh
